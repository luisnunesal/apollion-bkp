Hook para copiar conteudo textual para a are√° de transferencia.

## Exemplo

```jsx
export function UseClipboardExample() {
  const [inputValue, setInputValue] = useState(
    "Super important text to long to type manually"
  );

  const { onCopy, hasCopied } = useClipboard(inputValue);

  return (
    <Flex gap="large">
      <TextArea
        size="compact"
        value={inputValue}
        onChange={(e) => setInputValue(e.target.value)}
      />

      <Button size="extraSmall" variant="outlined" onClick={onCopy}>
        {hasCopied ? "Copiado" : "Copiar"}
      </Button>

      <TextArea size="compact" placeholder="Cole Aqui" />
    </Flex>
  );
}
```
